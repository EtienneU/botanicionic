import { Component, OnInit } from '@angular/core';
import {MenuController} from '@ionic/angular';
import {Router} from '@angular/router';

@Component({
  selector: 'app-plants',
  templateUrl: './plants.page.html',
  styleUrls: ['./plants.page.scss'],
})
export class PlantsPage implements OnInit {

  public plantList;
  constructor(
    private menu: MenuController,
    private router: Router
  ) { }

  ngOnInit() {
    this.menu.enable(
      true,
      'first'
    );

    this.plantList = [
      { name: 'Ficus', likes: 32, id: 20055 },
      { name: 'Camellia', likes: 56, id: 20053 },
      { name: 'Aloe Vera', likes: 61, id: 45652 },
      { name: 'Albizia', likes: 161, id: 44822 },
      { name: 'Melia', likes: 57, id: 45659 }
    ];

    console.log('J\'arrive sur la page des plantes =)');
  }

  ionViewWillEnter(){
    console.log('retour sur la page plants');
  }

  onAddPlant() {
    this.router.navigate(['add-plant']);
  }
}
